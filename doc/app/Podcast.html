Parse the xml returned by iTunes. If the podcast exists, update attributes. Otherwise, create a new podcast.

 72:   def self.scrape_from_itunes(itunes_doc)
 73:     itunes_doc.xpath('//feed/entry').map do |entry|
 74:       new_name = entry.xpath("./name").text
 75:       podcast = Podcast.find_all_by_name(new_name)
 76:       if (podcast == [])
 77:         begin
 78:           podcast = Podcast.create(
 79:             :name => entry.xpath("./name").text,
 80:             :itunesurl => entry.xpath("./link/@href").text,
 81:             :category => entry.xpath("./category/@term").text,
 82:             :hosts => entry.xpath("./artist").text,
 83:             :description => entry.xpath("./summary").text,
 84:             :artwork => entry.xpath("./image[@height='170']").text  
 85:           )
 86:           puts "New Podcast: #{podcast.name}"
 87:           Podcast.podcast_logger.info("New Podcast: #{podcast.name}")
 88:         rescue StandardError => ex
 89:           puts "An error of type #{ex.class} happened, message is #{ex.message}"
 90:         end
 91:       elsif (!podcast[0].nil?)
 92:         begin
 93:           pod = podcast[0]
 94:           pod.itunesurl = entry.xpath("./link/@href").text
 95:           pod.category = entry.xpath("./category/@term").text
 96:           pod.hosts = entry.xpath("./artist").text
 97:           pod.description = entry.xpath("./summary").text

101:             pod.save

105:         rescue StandardError => ex

107:         end
